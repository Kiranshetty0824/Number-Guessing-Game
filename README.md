# Number-Guessing-Game
Number Guessing Game

import random

number = random.randint(1, 100)
attempts = 0

print("Guess the number between 1 and 100!")

while True:
    guess = int(input("Your guess: "))
    attempts += 1
    if guess < number:
        print("Too low!")
    elif guess > number:
        print("Too high!")
    else:
        print(f"ğŸ‰ Correct! You guessed it in {attempts} tries.")
        break
